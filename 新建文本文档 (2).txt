 #include<stdio.h>
#include<string.h>
int main(void)
{
	char a[]="1+2+2+1+2+5+4-1-3+4-8";
	int sum=0;
	for(int i=1;i<strlen(a);i+=2)
		if(a[i]=='+')
			sum=sum+(a[i+1]-'0');
		else if(a[i]=='-')
			sum=sum-(a[i+1]-'0');
	 int c;
	 c=a[0]-'0';
	 int x;
	 x=c+sum;
	 printf("%d",x);
	return 0;
}
