#include<stdio.h>
#include<string.h>
int main()
{
	char a[]="2*3*4-6+7-2*3";
	int sum=a[0]-'0';
	for(int i=1;i<strlen(a);i+=2)
	{
		if(a[i]=='*')
		{
			int b=a[i+1]-'0';
			for(i+=2;i<strlen(a);i+=2)
			{
				if(a[i]=='*')
				{
					b=b*(a[i+1]-'0');
				}
				if(a[i]=='/')
				{
					b=b/(a[i+1]-'0');
				}
				if(a[i]=='+'||a[i]=='-')
				{
					i-=2;
					break;
				}
				sum=sum*b;
			}
		}
		else if(a[i]=='/')
		{
			int b=a[i+1]-'0';
			for(i+=2;i<strlen(a);i+=2)
			{
				if(a[i]=='*')
				{
					b=b*(a[i+1]-'0');
				}
				if(a[i]=='/')
				{
					b=b/(a[i+1]-'0');
				}
				if(a[i]=='+'||a[i]=='-')
				{
					i-=2;
					break;
				}
				sum=sum/b;
			}
		}
		else if(a[i]=='+')
		{
			int b=a[i+1]-'0';
			for(i+=2;i<strlen(a);i+=2)
			{
				if(a[i]=='*')
				{
				    b=b*(a[i+1]-'0');
				}
				if(a[i]=='/')
				{
				    b=b/(a[i+1]-'0');
				} 
				if(a[i]=='+'||a[i]=='-')
				{
					i-=2;
					break;
				}
			}
			sum=sum+b;
		}
		else if(a[i]=='-')
		{
			int b=a[i+1]-'0';
			for(i+=2;i<strlen(a);i+=2)
			{
				if(a[i]=='*')
				{
				    b=b*(a[i+1]-'0');
				}
				if(a[i]=='/')
				{
				    b=b/(a[i+1]-'0');
				} 
			    if(a[i]=='+'||a[i]=='-')
				{
					i-=2;
					break;
				}
			}
			sum=sum-b;
		}
	 } 
	 printf("%d",sum);
	 return 0;
}
